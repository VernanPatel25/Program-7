class ASCII
{
    public void ASCII()
    {
        char cha = 'A', chz = 'Z', chA = 'a', chZ = 'z';
        int asciiA, asciiZ, asciia, asciiz;
        asciiA = (int)cha;
        asciiZ = (int)chz;
        asciia = (int)chA;
        asciiz = (int)chZ;
        System.out.println("ASCII code for A: "+asciiA);
        System.out.println("ASCII code for a: "+asciia);
        System.out.println("ASCII code for Z: "+asciiZ);
        System.out.println("ASCII code for z: "+asciiz);
    }
}
